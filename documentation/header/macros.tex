\newcommand{\FG}[5]
{ 
	\vspace{2mm}
	 
	\begin{figure}[!htb]
	
		\centering
		
			\includegraphics[scale = #2]{#1} 	
			\caption[#3]{#4}
			\label{pic:#5}
			 
	\end{figure}
}

% ... und Referenz
\newcommand{\FR}[1]{\figurename~\ref{pic:#1}}
